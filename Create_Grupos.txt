CREATE ROLE grupo_funcionarios;

GRANT USAGE ON SCHEMA public TO grupo_funcionarios;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO grupo_funcionarios;

REVOKE UPDATE ON tb_funcionarios FROM grupo_funcionarios;
REVOKE CREATE ON tb_produtos FROM grupo_funcionarios;

ALTER ROLE grupo_funcionarios WITH CREATEROLE;
